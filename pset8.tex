\question Prove that if $A, B$ and $C$ are sets then $$A - (B \cap C) = (A - B) \cup (A - C).$$ 

\begin{solution}\begin{proof}
    Let $x \in A-(B \cap C)$. Then $x \in A$ and $x \notin (B \cap C)$, so
    $x \notin B$ or $x \notin C$. Then, $x \in (A - B)$ or $x \in (A - C)$, so $x \in (A-B) \cup (A-C)$.
    So $A-(B \cap C) \subseteq (A-B) \cup (A-C)$.

    Let $x \in (A-B) \cup (A-C)$. Then $x \in A-B$ or $x \in A-C$. In either case, $x \in A$.
    Also, we have that $x \notin B$ or $x \notin C$, so $x \notin (B \cap C)$. Then $x \in A-(B \cap C)$.
    So $(A-B) \cup (A-C) \subseteq A-(B \cap C)$.

    Since we showed that $A-(B \cap C) \subseteq (A-B) \cup (A-C)$ and
    $(A-B) \cup (A-C) \subseteq A-(B \cap C)$, then $A - (B \cap C) = (A - B) \cup (A - C)$.
\end{proof}\end{solution}


\question Prove that if $A, B$ and $C$ are sets then $$A \times (B \cup C) = (A \times B) \cup (A \times C).$$

\begin{solution}\begin{proof}
    Let $(x,y) \in A \times (B \cup C)$. Then $x \in A$ and $y \in B \cup C$, so $y \in B$ or $y \in C$.
    If $y \in B$ then $(x,y) \in A \times B$, and if $y \in C$ then $(x,y) \in A \times C$, so
    $(x,y) \in (A \times B) \cup (A \times C)$. So $A \times (B \cup C) \subseteq (A \times B) \cup (A \times C)$.

    Let $(x,y) \in (A \times B) \cup (A \times C)$. Then $x \in A$ and $y \in B$ or $y \in C$, so
    $y \in (B \cup C)$. Then $(x,y) \in A \times (B \cup C)$. So $(A \times B) \cup (A \times C)
    \subseteq A \times (B \cup C)$.

    Since we showed that $A \times (B \cup C) \subseteq (A \times B) \cup (A \times C)$ and
    $(A \times B) \cup (A \times C) \subseteq A \times (B \cup C)$, then we have that 
    $A \times (B \cup C) = (A \times B) \cup (A \times C)$.
\end{proof}\end{solution}


\question Prove that if $A$ and $B$ are sets then $A \subseteq B$ if and only if $A - B = \emptyset$.

\begin{solution}\begin{proof}
Assume that $A \subseteq B$. Assume for the sake of
contradiction that $A - B \ne \emptyset$. Then there exists some $x \in A - B$, so some $x \in A$
and $x \notin B$. This is a contradiction because $A \subseteq B$, so all $x \in A$ also satisfy $x \in B$.
Therefore if $A \subseteq B$, then $A - B = \emptyset$.

Assume for contrapositive that $A \nsubseteq B$. Then there exists some $a \in A$ such that $a \notin B$.
Consider $A-B$. $A-B$ must include $a$ because $a \in A$ and $a \notin B$. Since $a \in A-B$, then
$A-B \neq \emptyset$. Therefore, by contrapositive we have that if $A-B = \emptyset$, then $A \subseteq B$.

Since we have show that $A \subseteq B \Rightarrow A-B = \emptyset$ and $A-B = \emptyset \Rightarrow A
\subseteq B$, then we have that $A \subseteq B \iff A-B = \emptyset$.
\end{proof}\end{solution}

\question If $x,y \in \mathbb{R}$, then $|x + y| = |x| + |y|$. 

\begin{solution}\begin{proof}
False. Let $x = 1$ and $y = -1$. Then $|x+y| = |1-1| = 0 \neq |x| + |y| = |1| + |-1| = 2$.
\end{proof}\end{solution}


\question If there exists a set $A$ such that $A \cap B = A \cap C$ and $A \cup B = A \cup C$, then $B=C.$

\begin{solution}\begin{proof}
    True. Assume that there exists a set $A$ such that $A \cap B = A \cap C$ and $A \cup B = A \cup C$.

    Let $b \in B$. Either $b \in A$ or $b \notin A$.

    Case $b \in A$: Then $b \in B$ and $b \in A$ so $b \in A \cap B$. Since $A \cap B = A \cap C$,
    then $b \in A \cap C$, so $b \in C$. So $B \subseteq C$.

    Case $b \notin A$: Since $b \in B$, then $b \in A \cup B$. Since $A \cup B = A \cup C$,
    then $b \in A$ or $b \in C$. Since we know that $b \notin A$, then $b \in C$. So $B \subseteq C$.

    Similarly for $c \in C$. Either $c \in A$ or $c \notin A$. In either case, $C \subseteq B$.

    Since $B \subseteq C$ and $C \subseteq B$, then $B = C$.
\end{proof}\end{solution}


\question Let $A$ and $B$ sets with a universal set U. 
$$\overline{A \cap B} = \overline{A} \cup \overline{B}$$

\begin{solution}\begin{proof}
    True. Let $A$ and $B$ sets with a universal set U.
    \begin{align*}
        \overline{A \cap B} &= \{x \in U: x \notin A \cap B\}\\
        &= \{x \in U: x \notin A \lor x \notin B\}\\
        &= \{x \in U: x \notin A\} \cup \{x \in U: x \notin B\}\\
        \overline{A \cap B} &= \overline{A} \cup \overline{B}
    \end{align*}
\end{proof}\end{solution}

\question A number is ``deficient” if it is greater than the sum of its proper divisors. For example, 35 is deficient, since 1 + 5 + 7 = 13, which is less than 35. A number is called “abundant” if it is less than the sum of its proper divisors. For example, 36 is abundant, since 1 + 2 + 3 + 4 + 6 + 9 + 12 + 18 = 55, which is greater than 36. Are prime powers (i.e numbers of the form $x=p^k$ where $p$ is prime) deficient or abundant?

\begin{solution}\begin{proof}
Let $x$ be a prime power. Then $x = p^k$, where $p$ is prime. Then the factors of $x$ are
$p^0=1, p, p^2, ..., p^k=x$. Then the sum of the proper divisors of $x$ is $\sum_{n=0}^{k-1} p^k = 
\frac{1-p^k}{1-p} = \frac{p^k - 1}{p-1}$. Then $\frac{p^k-1}{p-1} < p^k-1 < p^k$, so prime powers are
deficient.
\end{proof}\end{solution}

\question We want to show that $f(x)=x^2-3x-2$ has a unique zero between 3 and 4. 
\begin{parts}
\item Explain why a zero exists using the intermediate value theorem. 

\begin{solution}
    $f(x)$ is a polynomial so it is continuous on $\mathbb{R}$, and since $[3, 4] \subseteq \mathbb{R}$,
    then $f(x)$ is continuous on $[3, 4]$.
    \begin{align*}
        f(3) &= 3^2 - 3(3) - 2 = -2 \\
        f(4) &= 4^2 - 3(4) - 2 = 2
    \end{align*}

    Since $0 \in [f(3), f(4)] = [-2, 2]$ and
    $f(x)$ is continuous on $[3, 4]$, then by the IVT there must exist some $c \in [3, 4]$ such that
    $f(c) = 0$.
\end{solution}


\item Assume $3<a<b<4$. Explain why $a^2-3a < b^2-3b$.  (Hint: if $w<x$ and $y<z$, when is $wy<xz$?)

\begin{solution}
Since $a < b$, then $a-3 < b-3$. Since $3 < a$, then $a-3 > 0$ and $0 < a-3 < b-3$.
Then $(a-3)a < (b-3)b$, so $a^2 - 3a < b^2 - 3b$.
\end{solution}


\item Explain why $f(a)<f(b)$. Use this to conclude why there must be a unique $x$ between 3 and 4 such that $f(x)=0$. 

\begin{solution}
Since $a^2 - 3a < b^2 - 3b$, then $a^2 - 3a - 2 < b^2 - 3b -2$, so $f(a) < f(b)$.

Assume that $x \in [3, 4]$ and $f(x) = 0$. Assume for the sake of contradiction that there exists some
$y \in [3,4]$, such that $y \ne x$ and $f(y) = 0$. Then either $x < y$ or $y < x$.

If $x < y$, then $f(x) < f(y)$, so $0 < f(y)$, so $f(y) \ne 0$. This is a contradiction.

If $y < x$, then $f(y) < f(x)$, so $f(y) < 0$, so $f(y) \ne 0$. This is a contradiction.

So we know that if there exists an $x \in [3, 4]$ such that $f(x) = 0$, then that $x$ is unique.

Since we know by IVT that there exists a $x \in [3, 4]$ such that $f(x) = 0$ and that if there is some
$x \in [3, 4]$ such that $f(x) = 0$ then it must be unique, then we know that there must be a unique
$x \in [3, 4]$ such that $f(x) = 0$.
\end{solution}
\end{parts}

\question Prove that $$\bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2 ] =[3,5].$$
\begin{solution}\begin{proof}
    First we show that $\bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2 ] \subseteq [3,5]$ by contrapositive.

    Let $a \notin [3, 5]$. Since $[3, 5] = [3-x^2, 5+x^2]$ for $x = 0$, then
    $\bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2] = [3, 5] \cap \bigcap_{x \in \mathbb{R}-\{0\}}$.
    Since $a \notin [3, 5]$, then $a \notin [3, 5] \cap \bigcap_{x \in \mathbb{R}-\{0\}}$
    and $a \notin \bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2]$.

    Then, by contrapositive, we have that $\bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2 ] \subseteq [3,5]$.\\


    Now we show that $[3, 5] \subseteq \bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2]$.

    Let $a \in [3, 5]$. Then $3 \le a \le 5$. Since $3-x^2 \le 3$ and $5 \le 5+x^2$ for all
    $x \in \mathbb{R}$, then $3-x^2 \le a \le 5+x^2$ for all $x \in \mathbb{R}$, so
    $a \in [3-x^2, 5+x^2]$ for all $x \in \mathbb{R}$. Then $a \in \bigcup_{x \in \mathbb{R}} [3-x^2, 5+x^2]$,
    and we have that $[3, 5] \subseteq \bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2]$.

    Since we showed that $\bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2 ] \subseteq [3,5]$
    and $[3, 5] \subseteq \bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2]$, then we have that
    $\bigcap_{x \in \mathbb{R}} [3-x^2, 5+x^2 ] = [3,5]$.
\end{proof}\end{solution}


\question Find a necessary and sufficient condition for $(A \times B) \cap (B \times A) = \emptyset.$  Verify the condition you found is necessary and sufficient. 
\begin{solution}
    $A \cap B = \emptyset \iff (A \times B) \cap (B \times A) = \emptyset$.

    Assume that $A \cap B = \emptyset$. Assume for the sake of contradiction that
    $(A \times B) \cap (B \times A) \neq \emptyset$. Then there exists some
    $(x, y) \in (A \times B) \cap (B \times A)$. Then $x \in A$ and $x \in B$, and $y \in B$ and $y \in A$,
    so $x \in A \cap B$ and $y \in A \cap B$. This is a contradiction because we assumed that
    $A \cap B = \emptyset$ but found an element in $A \cap B$.
    So we have that if $A \cap B = \emptyset$, then $(A \times B) \cap (B \times A) = \emptyset$.

    Assume that $(A \times B) \cap (B \times A) = \emptyset$. Assume for the sake of contradiction that
    $A \cap B \ne \emptyset$. Then there must exist some $x \in A \cap B$.
    Since $x \in A \cap B$, then $x \in A$ and $x \in B$. Then $(x, x) \in A \times B$ and
    $(x,x) \in B \times A$. Then $(x, x) \in (A \times B) \cap (B \times A)$. This is a contradiction
    because we assumed that $(A \times B) \cap (B \times A) = \emptyset$ but found an element
    $(x,x) \in (A \times B) \cap (B \times A)$.
    So we have that if $(A \times B) \cap (B \times A) = \emptyset$, then $A \cap B = \emptyset$.

    Therefore $A \cap B = \emptyset \iff (A \times B) \cap (B \times A) = \emptyset$.
\end{solution}


\question Prove or disprove: If $A \cup B = A \cup C$ for every set $A$, then $B=C.$
\begin{solution}\begin{proof}
    True. Assume that $A \cup B = A \cup C$ for every set $A$. Let $A = \emptyset$. Then
    $\emptyset \cup B = \emptyset \cup C$, so $B = C$.
\end{proof}\end{solution}


\question Suppose that $n$ is an odd number and $M=2n$ is perfect. Show that $n$ must equal 3.
\begin{solution}\begin{proof}
    Assume that $n$ is odd and $M = 2n$ is perfect. Then $M$ is an even perfect number, so
    $M = 2^{k-1} (2^k - 1)$ for some $k \in \mathbb{Z}$.
    Then $n = 2^{k-2} (2^k - 1)$. Since $n$ must be odd, it cannot have $2$ as a factor, so
    $2^{k-2} = 1 = 2^0$, so $k = 2$. Then $n = 2^0 (2^2 - 1) = 3$.
\end{proof}\end{solution}
