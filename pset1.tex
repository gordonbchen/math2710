\question Determine whether each of the following is true or false. Explain your reasoning for each answer:
	\begin{parts}
		\part The empty set is a proper subset of every set.

        \begin{solution} \\
		False, $\emptyset$ is not a proper subset of $\emptyset$ because $\emptyset = \emptyset$.
        \end{solution}

		\part The empty set is an element of every set.

        \begin{solution} \\
		False, $\emptyset$ is not an element of the set $\{1, 2, 3\}$.
        \end{solution}

		\part	$\{1\} \in \{1,2,3,\{4\}\}$

        \begin{solution} \\
		False, the set $\{1\}$ is not an element of $\{1,2,3,\{4\}\}$.
        \end{solution}

		\part $\{1\} \subseteq \{1,2,3,\{4\}\}$

        \begin{solution} \\
		True, $\{1\}$ is a subset of $\{1,2,3,\{4\}\}$ because every element of $\{1\}$, i.e. 1, is in $\{1,2,3,\{4\}\}$.
        \end{solution}

		\part $\{4\} \subseteq \{1,2,3,\{4\}\}$

        \begin{solution} \\
		False, the set $\{4\}$ is not a subset of $\{1,2,3,\{4\}\}$ because 4 is not in $\{1,2,3,\{4\}\}$.
        \end{solution}

		\part $\{3\} \subset \mathcal{P}(\mathbb{Q})$

        \begin{solution} \\
		False, $\{3\}$ is not a proper subset of $\mathcal{P}(\mathbb{Q})$ because 3 is not a subset of $\mathbb{Q}$.
        \end{solution}

		\part $\{\{\emptyset\}\} \subseteq \mathcal{P}(\{\emptyset, \{\emptyset\}\})$

        \begin{solution} \\
            True, $\{\{\emptyset\}\}$ is a subset of $\mathcal{P}(\{\emptyset, \{\emptyset\}\})$ because every element of $\{\{\emptyset\}\}$, i.e. $\{\emptyset\}$, is a subset of $\{\emptyset, \{\emptyset\}\}$.
        \end{solution}
		
	\end{parts}
	
\question Give an example, if possible, of sets $A,B,$ and $C$ satisfying the following. If it is not possible, explain why not.
	\begin{parts}
		\part $A\subseteq B$, $B \subseteq C$, and $C \subseteq A$

        \begin{solution} \\
		$A = B = C = \{1\}$
        \end{solution}

		\part $A\subset B$, $B \subset C$, and $C \subset A$

        \begin{solution} \\
            Impossible. By the definition of a proper subset, we have $A \subseteq B, \ A \neq B, \ B \subseteq C, \ B \neq C, \ C \subseteq A, \ C \neq A$. Since $A \subseteq B, \ B \subseteq C$, we have $A \subseteq C$. Then we have $A \subseteq C, \ C \subseteq A$, so $A = C$, but we also have $A \neq C$, so we have a contradiction. Therefore, $A\subset B$, $B \subset C$, and $C \subset A$ is impossible.
        \end{solution}

		\part $A \subseteq B$, $B \nsubseteq C$, and $C \nsubseteq A$

        \begin{solution} \\
		$A = \{1\}, B = \{1, 2\}, C = \{3\}$
        \end{solution}

	\end{parts}

\end{solution}

\question Assume $A$ and $B$ are sets with $|A|=4$ and $|B|=3.$ Determine $|A \times B|$. Explain your reasoning. 

\begin{solution} \\
$|A \times B| = |A| \times |B| = 12$. $|A \times B| = |A| \times |B|$ because there are $|A|$ choices for the first element of the tuples in $A \times B$, $|B|$ choices for the second element of the tuples in $A \times B$, and the first and second elements are chosen independently so there are $|A| \times |B|$ choices in total.
\end{solution}

\question We will see the following two definitions later in the course, but you are probably already familiar with them.
\begin{definition} An integer $n$ is \emph{even} if $n=2a$ for some integer $a \in \mathbb{Z}$. \end{definition}
\begin{definition} An integer $n$ is \emph{odd} if $n=2a+1$ for some integer $a \in \mathbb{Z}$. \end{definition}
Read the proofs below. What is being proven? Are there any gaps or holes in the logic?  The first one is well-written while the second is not. What are some things that stand out as bad in the second proof?

\begin{proof} Suppose the integer $x$ is odd. Then $x = 2a+1$ for some $a \in \mathbb{Z}$, by definition
of an odd number. Thus $x^2 = (2a+1)^2 = 4a^2+4a+1 = 2(2a^2+2a)+1$, so
$x^2 = 2b +1$ where $b = 2a^2 +2a \in  \mathbb{Z}$. Therefore $x^2$ is odd, by definition
of an odd number. \end{proof}

\begin{proof} $x = 2k + 1.$
Square it so
\[
x^2 = (2k+1)^2 = 4k^2 + 4k + 1.
\]
That's even, plus one. That's odd. {So it works.}
\end{proof}

\begin{solution} \\
The following is being proven: If $x$ is odd, then $x^2$ is odd.

The first proof does not have holes in the logic. The second proof does: the proof does not explicitly say $k \in \mathbb{Z}$ or put $4k^2 + 4k + 1$ into the form $2b + 1$.

The sentences in the second proof are incomplete, it's unclear what "that" refers to, and the proof does not show that $4k^2 + 4k + 1$ can be put into the form $2b + 1, b \in \mathbb{Z}$.
\end{solution}

\question Adapt the good proof above (in problem 9) to prove a complimentary statement about even numbers.

Show that if $x$ is even, then $x^2$ is even.

\begin{solution} \\
\begin{proof} Suppose the integer $x$ is even. Then $x = 2a$ for some $a \in \mathbb{Z}$, by definition of an even number. Thus $x^2 = (2a)^2 = 4a^2$, so $x^2 = 2b$ where $b = 2a^2 \in \mathbb{Z}$. Therefore $x^2$ is even, by definition of an even number. \end{proof}
\end{solution}
