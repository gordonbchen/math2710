 \question Define relations $f$ and $g$ on $\mathbb{R}$ by $f = \{ (x,y): (y=x) \vee (y=-x) \}$ and $g = \{ (x,y): y=x^2\}$. Prove that $f$ is not a function, but that $gf=\{(x,y):\exists z\in\mathbb{R}, (x,z)\in f \wedge (z,y)\in g \}$ is a function.

\begin{solution}

\begin{proof}
$(2, 2) \in f$ since $2 = 2$ and $(2, -2) \in f$ since $2 = -(-2)$. Then $f$ is not a function because
$2$ maps to 2 different outputs.
\end{proof}

\begin{proof}
Let $(x, y), (x, y_1) \in gf$.

Since $(x, y) \in gf$. Then there exists some $z \in \mathbb{R}$ such that $z = x$ or $z = -x$ and $y = z^2$.
If $z = x$, then $y = x^2$. If $z = -x$, then $y=(-x)^2 = x^2$. So in either case, $y = x^2$.

Since $(x, y_1) \in gf$. Then there exists some $z_1 \in \mathbb{R}$ such that
$z_1 = x$ or $z_1 = -x$ and $y_1 = z_1^2$.
If $z_1 = x$, then $y_1 = x^2$. If $z_1 = -x$, then $y_1 =(-x)^2 = x^2$. So in either case, $y_1 = x^2$.

Then $y = y_1$. Since $(x, y), (x, y_1) \in gf \implies y = y_1$, $gf$ maps each input to a unique output,
so it is a function.
\end{proof}

\end{solution}

 
 \question Let $A$ be a nonempty set. Prove that there exists an injective function from $\mathcal{P}(A)$ to $\mathcal{P}(\mathcal{P}(A))$. Is the function you found bijective?

\begin{solution}

\begin{proof}
Let $f = \{(a \in \mathcal{P}(A), p \in \mathcal{P}(\mathcal{P}(A))): p = \{a\}\}$.

Injective: we want to show that if $a \ne a'$, then $f(a) \ne f(a')$.

We proceed by contrapositive.

Assume that $f(a) = f(a')$. Then $p = \{a\} = p' = \{a'\}$, so $a = a'$.

Then by contrapositive we have that $a \ne a' \implies f(a) \ne f(a')$, so $f$ is injective.
\end{proof}

 \question Use mathematical induction to prove that if $A$ and $B$ are finite sets with $|A|=|B|=n$, then there are $n!$ bijective functions from $A$ to $B$.

\begin{solution}\begin{proof}
Base case $n=1$:

Let $A$ and $B$ be finite sets where $|A| = |B| = 1$. Then $A = {a}$ and $B = {b}$.
Then the only bijective function from $A$ to $B$ is $f = \{(a, b)\}$. Since there is only $n! = 1! = 1$
bijective function from $A$ to $B$, the base case holds.\\


Induction step:

Assume that there are $k!$ bijective functions between 2 sets with $k$ elements.

Consider 2 sets with $k+1$ elements $A = \{a_1 ... a_{k+1}\}$ and $B = \{b_1 ... b_{k+1}\}$.

For every $i \in {1... k+1}$, make a bijective function $f_i$ by taking the union between the
bijective function mapping $\{a_1 ... a_k\}$ to $\{b_j: j \ne i\}$ and $\{(a_{k+1}, b_i)\}$.

Since there are $k+1$ choices for $i$, and
$k!$ bijective functions mapping $\{a_1 ... a_k\}$ to $\{b_j: j \ne i\}$ for each choice of $i$,
then there are $k! (k+1)$ bijective functions mapping between 2 sets with $k+1$ elements.\\

Then by induction, we have that there are $n!$ bijections between finite sets with $n$ elements.

\end{proof}\end{solution}


\question Prove or disprove each of the following:
\begin{parts}
	\part There exist functions $f:S \rightarrow T$ and $g:T \rightarrow U$ such that $f$ is not surjective and $g\circ f$ is surjective. 

\begin{solution}\begin{proof}
Let $S = \{1\}$, and $T = \{2, 3\}$, and $U = \{4\}$. Let $f = \{(1, 2)\}$, and $g = \{(2, 4)\}$.

Then $f$ is not surjective because nothing maps to $3$.

Then $g \circ f = \{(1, 4)\}$. Then $g \circ f$ is surjective because for all $u \in U$, i.e. $4$,
there exists a $s \in S$, i.e. $1$, such that $g \circ f(1) = 4$.
\end{proof}\end{solution}

 
	\part There exist functions $f:S \rightarrow T$ and $g:T \rightarrow U$ such that $f$ is not injective and $g\circ f$ is injective. 

\begin{solution}\begin{proof}
False.

Assume for the sake of contradiction that there exists $f: S \to T$ and $g: T \to U$ such that
$f$ is not injective and $g \circ f$ is injective.

Since $f$ is not injective, then there must exist $x_1, x_2 \in S$ such that $f(x_1) = f(x_2)$ and
$x_1 \ne x_2$.

Since $f(x_1) = f(x_2)$, then $g(f(x_1)) = g(f(x_2))$ since $g$ is a function, so then
$g \circ f(x_1) = g \circ f(x_2)$.

Since $g \circ f$ is injective, then $x_1 = x_2$.
This is a contradiction, since we assumed that $x_1 \ne x_2$.

Then there cannot exist $f: S \to T$ and $g: T \to U$ such that
$f$ is not injective and $g \circ f$ is injective.
\end{proof}\end{solution}

 
\end{parts}

\begin{claim} The function $f:\mathbb{R}-\{1\} \rightarrow \mathbb{R}-\{3\}$ defined by $f(x)=\dfrac{3x}{x-1}$ is bijective. \end{claim}
 \begin{proof} 
 	First, we show that $f$ is one-to-one. Assume that $f(a)=f(b)$, where $a,b \in \mathbb{R}-\{1\}$. Then $\dfrac{3a}{a-1}=\dfrac{3b}{b-1}.$ Therefore, $3a(b-1)=3b(a-1)$, so $3ab-3a=3ab-3b$. From here, we see that $-3a=-3b$, and so $a=b$. This implies $f$ is injective.
	
	Now, to see that $f$ is onto, let $f(x)=r$. Then $\dfrac{3x}{x-1}=r$, so $3x=r(x-1)=rx-r.$ This implies that $3x-rx=-r$, so we have $x(3-r)=-r$. Finally, since $x\in \mathbb{R}-\{3\}$, we see that $x=\dfrac{-r}{3-r}=\dfrac{r}{r-3}$. Therefore, $$f(x)=f\left(\frac{r}{r-3}\right)=\frac{3\left(\frac{r}{r-3}\right)}{\frac{r}{r-3}-1}=\frac{3r}{r-(r-3)}=\frac{3r}{3}=r.$$ So $f$ is surjective.
 \end{proof}

\question Prove that if $a$ is a natural number, then there exist two unequal natural numbers $k$ and $l$ for which $a^k-a^l$ is divisible by $10.$

\begin{solution}\begin{proof}
Let $a \in \mathbb{N}$.

Consider the function $f: \mathbb{N} \to \{0...9\}$ where $f(n) = a^n \mod 10$.

Since $|\mathbb{N}| > |\{0...9\}|$, then by the pigeonhole principle $f$ cannot be injective.

Then there must exist $k, l \in \mathbb{N}$, such that $k \ne l$ and $f(k) = f(l)$.

Then $a^k \mod 10 = a^l \mod 10$, and we have that $a^k - a^l \mod 10 = 0$, so $10 | a^k - a^l$.

Then we have $k \ne l$ such that $10 | a^k - a^l$.
\end{proof}\end{solution}
