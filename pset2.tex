\question Let $\{a,b,c,d,e,f\}$ be the universal set, and let $R=\{a,b,c,d,f \}$, $S=\{b,c, f\}$, and $T=\{ d,e,f \}$.
	\begin{parts}
		\part $R \cap T$
        \begin{solution} \\
        $R \cap T = \{d, f\}$
        \end{solution}

		\part $R \cap (S\cup T)$
        \begin{solution} \\
        $R \cap (S \cup T) = R \cap \{b, c, d, e, f\} = \{b, c, d, f\}$
        \end{solution}

		\part $(R - S) - T$
        \begin{solution} \\
        $(R-S) - T = \{a, d\} - \{d, e, f\} = \{a\}$
        \end{solution}

		\part $R - (S - T)$
        \begin{solution} \\
        $R - (S - T) = R - \{b, c\} = \{a, d, f\}$
        \end{solution}

		\part $\mathcal{P}(\overline{R})$
        \begin{solution} \\
        $\mathcal{P}(\overline{R}) = \mathcal{P}(\{e\}) = \{\emptyset, \{e\}\}$
        \end{solution}
	\end{parts}

\question For each natural number $n$, let $A_n=\mathbb{N} - \{ 1,2,3, \ldots, n\}$. Determine $\bigcup_{n\in\mathbb{N} }A_n$ and $\bigcap_{n\in\mathbb{N}} A_n$.
\begin{solution}
\begin{proof} $\bigcup_{n \in \mathbb{N}} A_n = A_1$
\begin{lemma}
$A_{n+1} \subseteq A_n$

$A_{n+1} = \mathbb{N} - \{1, 2, 3, \ldots, n, n+1\} = A_n - \{n+1\}$

$A_n$ has all elements inside $A_{n+1}$ and the element $n+1$, so $A_{n+1} \subseteq A_n$.
\end{lemma}

Since $A_{n+1} \subseteq A_n$, $A_n \subseteq A_1$ for $n > 1$, and $\bigcup_{n \in \mathbb{N}} A_n = A_1$.
\end{proof}

\begin{proof}
$\bigcap_{n\in\mathbb{N}}A_n = \emptyset$

For sake of contradiction, assume $n' \in \bigcap_{n\in\mathbb{N}}A_n$. Consider the set $A_{n'} = \mathbb{N} - \{1, 2, \dots, n'\}$. Then $n' \notin A_{n'}$, and because $n' \in \mathbb{N}$, $n' \notin \bigcap_{n\in\mathbb{N}}A_n$. We assumed $n' \in \bigcap_{n\in\mathbb{N}}A_n$, but found that $n' \notin \bigcap_{n\in\mathbb{N}}A_n$, so we have a contradiction. Therefore $\bigcap_{n\in\mathbb{N}}A_n = \emptyset$.
\end{proof}

\end{solution}
